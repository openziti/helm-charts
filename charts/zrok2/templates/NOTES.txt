: this var must be exported so that the zrok2 CLI will configure itself
export ZROK2_API_ENDPOINT=$ZROK2_SCHEME://$ZROK2_HOST:$ZROK2_PORT

curl -sSf $ZROK2_API_ENDPOINT | grep '<title>'

Log in to the zrok2 console at $ZROK2_API_ENDPOINT using the credential obtained with this command:

kubectl -n {{ .Release.Namespace }} \
    get secrets {{ include "zrok2.fullname" . }}-ziggy-account-password \
    -o go-template='{{`{{"\n"}}{{range $k,$v := .data}}{{printf "%s: " $k}}{{if not $v}}{{$v}}{{else}}{{$v | base64decode}}{{end}}{{"\n"}}{{end}}`}}'

Enable your zrok2 account on a device that can reach the API by running:

ZROK2_ENABLE_TOKEN=$(kubectl -n {{ .Release.Namespace }} \
    get secrets {{ include "zrok2.fullname" . }}-ziggy-account-token \
    -o go-template='{{`{{index .data "token" | base64decode }}`}}'
)
zrok2 enable $ZROK2_ENABLE_TOKEN
